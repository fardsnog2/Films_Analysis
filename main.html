      text=f.read()
      s=text.decode('utf-8')
      js=json.loads(s)
      for feed in js['feed']:
        try:
          title=feed['meta']['title']
          if title==' ':
            msg=''
            for t in feed['payload']['body']:
              msg+=t['text']+' '
        except:
          continue
      ctg=js['reason']['category']['name']
      id_ctg=js['reason']['category']['id']
      df.loc[len(df)]=[msg.replace('X',''),ctg,id_ctg]
